const int VCC = 2;
const int TRIG = 3;
const int ECHO = 4;
const int GND = 5;

void setup() {
  pinMode(VCC, OUTPUT);
  pinMode(GND, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  digitalWrite(VCC, HIGH);
  long duration, inches;
  
  pinMode(TRIG, OUTPUT);
  digitalWrite(TRIG, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG, HIGH);
  delayMicroseconds(5);
  digitalWrite(TRIG, LOW);
  
  pinMode(ECHO, INPUT);
  duration = pulseIn(ECHO, HIGH);

  inches = microsecondsToInches(duration);

  Serial.print(inches);
  Serial.print(" in,");
  Serial.println();

  delay(1000); // Change this if needed
}

long microsecondsToInches(long microseconds) {
  return microseconds / 74 / 2;
}
